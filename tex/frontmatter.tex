% Portada con imagen (opcional). Coloca tu archivo en:
% - img/portada.jpg (recomendado) o img/portada.png o img/portada.pdf
% - o define \LibroPortadaArchivo en tex/metadatos.tex (ruta relativa).
\begingroup
\def\PortadaPath{}
\def\TryPortadaDefaults{%
  \IfFileExists{img/portada.jpg}{\def\PortadaPath{img/portada.jpg}}{%
    \IfFileExists{img/portada.png}{\def\PortadaPath{img/portada.png}}{%
      \IfFileExists{img/portada.pdf}{\def\PortadaPath{img/portada.pdf}}{}%
    }%
  }%
}
\milibroIfBlankTF{\LibroPortadaArchivo}{%
  \TryPortadaDefaults
}{%
  \IfFileExists{\LibroPortadaArchivo}{\def\PortadaPath{\LibroPortadaArchivo}}{\TryPortadaDefaults}%
}

\milibroIfBlankTF{\PortadaPath}{}{%
  \pdfbookmark[0]{Portada}{cover}
  \newgeometry{margin=0in}
  \thispagestyle{empty}
  \begin{center}
  \vspace*{\fill}
  \includegraphics[width=\paperwidth,height=\paperheight,keepaspectratio]{\PortadaPath}
  \vspace*{\fill}
  \end{center}
  \clearpage
  \restoregeometry
}
\endgroup
